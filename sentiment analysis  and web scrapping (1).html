# Import required libraries
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from textblob import TextBlob
import re
from nltk.corpus import stopwords
import nltk

# Download only essential NLTK data
try:
    nltk.download('stopwords')
    nltk.download('punkt')
except Exception as e:
    print(f"Error downloading NLTK resources: {e}")

# Set style for visualizations
sns.set_style('whitegrid')
plt.style.use('fivethirtyeight')

# Create sample tweets about Rwanda's development
tweets = [
    "Rwanda's economic growth is impressive! The government is doing great work with Vision 2050. #Rwanda #Development",
    "I'm concerned about the income inequality in Rwanda despite the economic growth numbers. #Rwanda",
    "Kigali is becoming a tech hub in Africa! So proud of Rwanda's progress. #DigitalTransformation",
    "The tourism sector in Rwanda is booming, especially with gorilla trekking. Amazing!",
    "Some rural areas in Rwanda still lack basic infrastructure while cities develop rapidly. #BalanceNeeded",
    "Rwanda's ease of doing business ranking improvement is remarkable! #Business",
    "Healthcare access has improved but more needs to be done in remote areas. #RwandaHealth",
    "The clean streets of Kigali show Rwanda's commitment to environmental sustainability. #CleanCity",
    "Unemployment among youth remains a challenge despite economic growth. #RwandaYouth",
    "Rwanda's digital transformation is setting an example for other African nations. #DigitalRwanda"
]

# Create DataFrame
df_tweets = pd.DataFrame({'tweet': tweets})

# Simplified clean text function that doesn't use punkt_tab
def clean_text(text):
    # Remove URLs, mentions, and hashtags
    text = re.sub(r'http\S+|www\S+|https\S+', '', text, flags=re.MULTILINE)
    text = re.sub(r'\@\w+|\#\w+', '', text)
    # Remove special characters and numbers
    text = re.sub(r'[^a-zA-Z\s]', '', text)
    # Convert to lowercase
    text = text.lower()
    # Simple word splitting (alternative to word_tokenize)
    words = text.split()
    # Remove stopwords
    stop_words = set(stopwords.words('english'))
    words = [word for word in words if word not in stop_words and len(word) > 1]
    return ' '.join(words)

# Apply cleaning
df_tweets['cleaned_tweet'] = df_tweets['tweet'].apply(clean_text)

# Sentiment analysis function
def get_sentiment(text):
    try:
        analysis = TextBlob(text)
        if analysis.sentiment.polarity > 0.1:
            return 'positive'
        elif analysis.sentiment.polarity < -0.1:
            return 'negative'
        else:
            return 'neutral'
    except:
        return 'neutral'

# Apply sentiment analysis
df_tweets['sentiment'] = df_tweets['cleaned_tweet'].apply(get_sentiment)
df_tweets['polarity'] = df_tweets['cleaned_tweet'].apply(lambda x: TextBlob(x).sentiment.polarity)
df_tweets['subjectivity'] = df_tweets['cleaned_tweet'].apply(lambda x: TextBlob(x).sentiment.subjectivity)

# Visualization
plt.figure(figsize=(10, 6))
sns.countplot(x='sentiment', data=df_tweets, order=['positive', 'neutral', 'negative'])
plt.title('Sentiment Distribution in Rwanda Development Tweets')
plt.savefig('sentiment_distribution.png')
plt.show()

plt.figure(figsize=(10, 6))
sns.scatterplot(x='polarity', y='subjectivity', hue='sentiment', data=df_tweets)
plt.title('Sentiment Analysis: Polarity vs Subjectivity')
plt.savefig('polarity_subjectivity.png')
plt.show()

# Save results
df_tweets.to_csv('rwanda_sentiment_analysis.csv', index=False)
print("Sentiment analysis completed successfully!")
print("Results saved to:")
print("- rwanda_sentiment_analysis.csv")
print("- sentiment_distribution.png")
print("- polarity_subjectivity.png")

# Import required libraries
import pandas as pd
import requests
from bs4 import BeautifulSoup

def scrape_quotes():
    base_url = "http://quotes.toscrape.com"
    current_page = "/"
    quotes = []
    authors = []
    tags_list = []
    
    while current_page:
        try:
            response = requests.get(base_url + current_page)
            response.raise_for_status()  # Check for HTTP errors
            soup = BeautifulSoup(response.text, 'html.parser')
            quote_divs = soup.find_all('div', class_='quote')
            
            for quote in quote_divs:
                quotes.append(quote.find('span', class_='text').get_text())
                authors.append(quote.find('small', class_='author').get_text())
                tags = [tag.get_text() for tag in quote.find_all('a', class_='tag')]
                tags_list.append(', '.join(tags))
            
            next_button = soup.find('li', class_='next')
            current_page = next_button.find('a')['href'] if next_button else None
        
        except Exception as e:
            print(f"Error during scraping: {e}")
            break
    
    df = pd.DataFrame({
        'Quote': quotes,
        'Author': authors,
        'Tags': tags_list
    })
    
    return df

# Execute scraping
try:
    quotes_df = scrape_quotes()
    print("First 5 quotes scraped:")
    print(quotes_df.head())
    
    # Save to CSV
    quotes_df.to_csv('quotes.csv', index=False)
    print("Web scraping completed. Data saved to quotes.csv")
except Exception as e:
    print(f"Error in web scraping: {e}")
